- query() 是 AsyncGenerator — 必须 drain：`for await (const _ of gen) {}`，之后 Terminal = gen.return(undefined).value 或从 done result 取
- querySource 必须用 'repl_main_thread'（非 'compact'/'session_memory'）

Drain helper 模式：
```typescript
async function drainQuery(gen: AsyncGenerator<unknown, unknown>): Promise<unknown> {
  let result
  for await (const _ of gen) { /* drain yielded events */ }
  // generator 自然完成后 Terminal 通过 done value 获取
  // 注意：for-await 结束后 gen 已关闭，使用 gen.return() 不再有效
  // 正确做法：在 for-await 中捕获最后一个 done value
  // 或使用 while 循环：
  // let done = false
  // while (!done) { const r = await gen.next(); if (r.done) { result = r.value; done = true } }
  return result
}
```

```typescript
import type { QueryDeps } from '../deps.js'
import type { Message } from 'src/types/message.js'

// Drain async generator and return the Terminal value
async function drainQuery(
  gen: ReturnType<typeof query>
): Promise<{ events: unknown[]; terminal: unknown }> {
  const events: unknown[] = []
  let terminal: unknown
  while (true) {
    const { value, done } = await gen.next()
    if (done) {
      terminal = value
      break
    }
    events.push(value)
  }
  return { events, terminal }
}

// Minimal QueryParams builder
function makeParams(overrides: Partial<Parameters<typeof query>[0]> = {}) {
  return {
    messages: [],
    systemPrompt: '' as any,  // asSystemPrompt('') 若 asSystemPrompt 可导入
    userContext: {},
    systemContext: {},
    canUseTool: async () => ({ result: 'allow' as const }),
    toolUseContext: createTestToolUseContext(),
    querySource: 'repl_main_thread' as const,
    maxTurns: 2,  // 防止意外无限循环
    ...overrides,
  }
}
```
